50  static std::vector<std::string> ListFiles(const std::string& path) {
51  const int fd = open(path.c_str(), O_DIRECTORY);
52  if (fd == -1) {
53  throw lucene::core::util::IOException(strerror(errno));
54  }
55 
56  std::vector<std::string> ret;
57  DIR* dirp = fdopendir(fd);
58  struct dirent* dp;
59 
60  while ((dp = readdir(dirp)) != NULL) {
61  if (strcmp(dp->d_name, ".") == 0 ||
62  strcmp(dp->d_name, "..") == 0) {
63  continue;
64  }
65 
66  ret.push_back(std::string(dp->d_name));
67  }
68 
69  closedir(dirp);
70  close(fd);
71 
72  return ret;
73  }

83  static void CreateDirectories(const std::string& path) {
84  struct stat statbuf;
85  char path_buf[PATH_MAX + 1];
86  std::memcpy(path_buf, path.c_str(), path.length());
87  path_buf[path.length()] = '\0';
88  char* ptr = path_buf;
89  bool done = false;
90 
91  while (!done) {
92  ptr += strspn(ptr, "/");
93  ptr += strcspn(ptr, "/");
94  done = (*ptr == '\0');
95  *ptr = '\0';
96 
97  if (stat(path_buf, &statbuf) == -1) {
98  if (errno == ENOENT) {
99  FileUtil::CreateDirectory(std::string(path_buf));
100  }
101  } else if (!S_ISDIR(statbuf.st_mode)) {
102  throw lucene::core::util::IOException(std::string(path_buf) +
103  " is not a directory");
104  }
105 
106  *ptr = '/';
107  }
108  }
