138 class BufferedChecksum: public lucene::core::util::Checksum {
139  public:
140  const uint32_t DEFAULT_BUFFERSIZE = 256;
141 
142  private:
143  std::unique_ptr<lucene::core::util::Checksum> in;
144  std::unique_ptr<char[]> buffer;
145  uint32_t buffer_size;
146  uint32_t upto;
147 
148  private:
149  void Flush() {
150  if (upto > 0) {
151  in->Update(buffer.get(), 0, upto);
152  upto = 0;
153  }
154  }
155 
156  public:
157  explicit BufferedChecksum(std::unique_ptr<lucene::core::util::Checksum>&& in)
158  : lucene::core::util::Checksum(),
159  in(std::forward<std::unique_ptr<lucene::core::util::Checksum>>(in)),
160  buffer(std::make_unique<char[]>(BufferedChecksum::DEFAULT_BUFFERSIZE)),
161  buffer_size(BufferedChecksum::DEFAULT_BUFFERSIZE),
162  upto(0) {
163  }
164 
165  BufferedChecksum(std::unique_ptr<lucene::core::util::Checksum>&& in,
166  const uint32_t buffer_size)
167  : lucene::core::util::Checksum(),
168  in(std::forward<std::unique_ptr<lucene::core::util::Checksum>>(in)),
169  buffer(std::make_unique<char[]>(buffer_size)),
170  buffer_size(buffer_size),
171  upto(0) {
172  }
173 
174  void Update(const char b) {
175  if (upto == buffer_size) {
176  Flush();
177  }
178 
179  buffer[upto++] = b;
180  }
181 
182  void Update(const char bytes[], const uint32_t off, const uint32_t len) {
183  if (len >= buffer_size) {
184  Flush();
185  in->Update(bytes, off, len);
186  } else {
187  if (upto + len > buffer_size) {
188  Flush();
189  }
190 
191  std::memcpy(buffer.get() + upto, bytes + off, len);
192  upto += len;
193  }
194  }
195 
196  int64_t GetValue() {
197  Flush();
198  return in->GetValue();
199  }
200 
201  void Reset() {
202  upto = 0;
203  in->Reset();
204  }
205 };
