335  bool IncrementToken() {
336  if (used) {
337  return false;
338  }
339 
340  ClearAttributes();
341  term_attr->Append(value);
342  offset_attr->SetOffset(0, value.size());
343  return (used = true);
344  }
345 
346  void End() {
347  lucene::core::analysis::TokenStream::End();
348  const uint32_t final_offset = value.size();
349  offset_attr->SetOffset(final_offset, final_offset);
350  }

353  used = true;
354  }

356  void Close() noexcept {
357  // In Java, it assigns a null to value but we do nothing here.
