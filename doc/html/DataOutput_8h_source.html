171  void CopyBytes(DataInput& input, const uint64_t num_bytes) {
172  AllocateCopyBufferIf();
173 
174  uint64_t left = num_bytes;
175  while (left > 0) {
176  const uint32_t to_copy =
177  (left > COPY_BUFFER_SIZE ? COPY_BUFFER_SIZE : left);
178 
179  // TODO(0ctopus13prime): Directly copy from input to
180  // this without buffer copying?
181  input.ReadBytes(copy_buffer.get(), 0, to_copy);
182  WriteBytes(copy_buffer.get(), 0, to_copy);
183  left -= to_copy;
184  }
185  }

280 
281  public:

283  : bytes(std::make_unique<char[]>(cp)),
284  bytes_len(cp),
285  length(0) {

288  void WriteByte(const char b) {
289  if (length >= bytes_len) {
290  std::pair<char*, uint32_t> result =
291  lucene::core::util::arrayutil::Grow(bytes.get(), bytes_len);
292  if (result.first != nullptr) {
293  bytes.reset(result.first);
294  bytes_len = result.second;
295  }
296  }
297 
298  bytes[length++] = b;
299  }
300 
301  void WriteBytes(const char in_bytes[],
302  const uint32_t offset,
303  const uint32_t in_length) {
304  const uint32_t new_length = length + in_length;
305  if (new_length >= bytes_len) {
306  std::pair<char*, uint32_t> result =
307  lucene::core::util::arrayutil::Grow(bytes.get(), new_length);
308  if (result.first != nullptr) {
309  bytes.reset(result.first);
310  bytes_len = result.second;
311  }
312  }
313 
314  std::memcpy(bytes.get() + length, in_bytes + offset, in_length);
315  length = new_length;
316  }

336 class FileIndexOutput: public IndexOutput {
337  public:
338  static const uint32_t BUF_SIZE = 8192;
339 
340  private:
341  lucene::core::util::Crc32 crc;
342  uint64_t bytes_written;
343  std::string path;
344  int fd;
345  uint32_t buf_idx;
346  bool flushed_on_close;
347  char buffer[BUF_SIZE];
348 
349  private:
350  void flush() {
351  if (buf_idx > 0) {
352  write(fd, buffer, buf_idx);
353  buf_idx = 0;
354  }
355  }
356 
357  public:
358  FileIndexOutput(const std::string& resource_desc,
359  const std::string& name,
360  const std::string& path)
361  : IndexOutput(resource_desc, name),
362  crc(),
363  bytes_written(0L),
364  path(path),
365  fd(open(path.c_str(),
366  O_CREAT | O_WRONLY | O_EXCL,
367  S_IRWXU | S_IRGRP | S_IXGRP | S_IROTH | S_IXOTH)),
368  buf_idx(0),
369  flushed_on_close(false) {
370  if (fd < 0) {
371  throw lucene::core::util::IOException(std::strerror(errno));
372  }
373  }
374 
375  ~FileIndexOutput() {
376  try {
377  Close();
378  } catch(...) {
379  // Ignore
380  }
381  }
382 
383  void WriteByte(const char b) {
384  crc.Update(b);
385  buffer[buf_idx++] = b;
386  bytes_written++;
387 
388  if (buf_idx >= BUF_SIZE) {
389  flush();
390  }
391  }
392 
393  void WriteBytes(const char bytes[],
394  const uint32_t offset,
395  const uint32_t length) {
396  bytes_written += length;
397  crc.Update(bytes, offset, length);
398 
399  if (length > BUF_SIZE) {
400  flush();
401  write(fd, bytes + offset, length);
402  return;
403  }
404 
405  if (length + buf_idx > BUF_SIZE) {
406  flush();
407  }
408 
409  std::memcpy(buffer + buf_idx, bytes + offset, length);
410  buf_idx += length;
411  }
412 
413  void Close() {
414  if (!flushed_on_close) {
415  flushed_on_close = true;
416  flush();
417 
418  const int result = close(fd);
419  if (result < 0) {
420  throw lucene::core::util::IOException(std::strerror(errno));
421  }
422  }
423  }
424 
425  uint64_t GetFilePointer() {
426  return bytes_written;
427  }
428 
429  uint64_t GetChecksum() {
430  if (!flushed_on_close) {
431  write(fd, buffer, buf_idx);
432  fsync(fd);
433  buf_idx = 0;
434  }
435 
436  return crc.GetValue();
437  }
438 };
