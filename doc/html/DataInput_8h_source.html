490  void ReadBytes(char bytes[],
491  const uint32_t offset,
492  const uint32_t len,
493  const bool use_buffer) {
494  const uint32_t available = buffer_length - buffer_position;
495  if (len <= available) {
496  std::memcpy(bytes + offset, buffer.get() + buffer_position, len);
497  buffer_position += len;
498  } else {
499  uint32_t offset_cp = offset;
500  uint32_t len_cp = len;
501 
502  if (available > 0) {
503  std::memcpy(bytes + offset, buffer.get() + buffer_position, available);
504  offset_cp += available;
505  len_cp -= available;
506  buffer_position += available;
507  }
508 
509  if (use_buffer && len_cp < buffer_size) {
510  Refill();
511  if (buffer_length < len_cp) {
512  std::memcpy(bytes + offset, buffer.get(), buffer_length);
513  throw lucene::core::util::EOFException();

515  std::memcpy(bytes + offset, buffer.get(), len_cp);
516  buffer_position = len_cp;
517  }
518  } else {
519  const uint64_t after = buffer_start + buffer_position + len_cp;
520  if (after > Length()) {
521  throw lucene::core::util::EOFException();
522  }
523 
524  ReadInternal(bytes, offset, len_cp);
