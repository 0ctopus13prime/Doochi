114 template<typename ATTR_FACTORY, typename ATTR_IMPL>
115 class AttributeFactory::StaticImplementationAttributeFactory
116  : public AttributeFactory {
117  private:
118  ATTR_FACTORY delegate;
119  static std::unordered_set<std::type_index> DEFAULT_ATTR_TYPE_IDS;
120 
121  public:
122  StaticImplementationAttributeFactory() { }
123  virtual ~StaticImplementationAttributeFactory() { }
124  AttributeImpl*
125  CreateAttributeInstance(const std::type_index attr_type_id) override {
126  ATTR_IMPL attr_impl;
127  std::vector<std::type_index> attributes = attr_impl.Attributes();
128  auto it = std::find(attributes.begin(), attributes.end(), attr_type_id);
129  if (it != attributes.end()) {
130  return new ATTR_IMPL();
131  } else {
132  return delegate.CreateAttributeInstance(attr_type_id);
133  }
134  }
135 };
