86 template <typename T>
87 T* CopyOfRange(const T* original, const uint32_t from, const uint32_t to) {
88  if (from >= to) {
89  throw std::invalid_argument("Range must guarantee from < to, from -> "
90  + std::to_string(from) + ", to -> " + std::to_string(to));
91  }
92 
93  const uint32_t new_length = (to - from);
94  T* copy = new T[new_length];
95  std::memcpy(copy, original + from, sizeof(T) * new_length);
96  return copy;
97 }

117  const INTEGER3 length) {
118  // Checks if `from_index` in [0, from_index + size]
119  // and `from_index` + `size` < = `length`
120  int64_t end = from_index + size;
121  if (from_index < 0 || from_index > end || end > length) {
122  throw std::invalid_argument(

124  + std::to_string(from_index) + " + " + std::to_string(size)
